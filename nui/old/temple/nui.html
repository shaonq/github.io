options.width:string 弹出层的宽高 ,宽："100px",高:["100px","80px"]
options.success:function(that,index) that:弹出层的jQ对象 index:弹出层的索引

    $('#animate_demo').each(function(){
        var that =$(this),btns =that.parent().next().find('a');
        btns.on("click",function () {
            var self =$(this),a ='is-disabled',b='readonly';
            if(this.className === "nui-btn"){
                btns.each(function (index,item) {
                    item.className = "nui-btn"
                }).addClass(a);
                that.removeAttr(b);
                this.className = "nui-btn nui-btn-green";
                var data =[];
                nui.animate(0, 10,function(value,end){
                    that.val(value);
                    data.push(value);
                    if (end) {
                        nui.open({
                            title:self.html(),
                            type:2,
                            width:["450px","300px"],
                            content:'../../mod/chart/?data='+JSON.stringify(data)
                        });
                        that.attr(b,b);
                        btns.removeClass(a);
                    }
                },self.html(),1e3)
            }
        })
    })
