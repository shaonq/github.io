// 上传
nui.upload({
        elem: "#elem",
        url: '../../nui/doc/success.json',
        ext: 'jpg',
        method: "get",
        error: function (res) {
            nui.toast('上传失败')
        },
        before: function (t) {
           nui.showLoading('上传中')
        },
        success: function (res, t) {
           nui.showSuccess('上传成功');
           $("#photo").val(res.msg);
        }
});
// 表单
nui.form({
        "form": ".nui-form", //必填，要渲染的UI
        "url":  '../../nui/doc/success.json', //请求地址，非form必填
        "btn": ".nui-btn-width", //触发form的submit  非form必填
        "message": function (name, that) {
                if(name === "photo"){
                nui.toast("请上传图片");
                }else{
                nui.toast("请输入");
                }
        },
    }, function (e) {   // 渲染UI，渲染事件
        nui.showSuccess(e.msg)
    },{type:'get'});
